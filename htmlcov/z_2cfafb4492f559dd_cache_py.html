1"""Multi-level caching system for PawControl integration. 

2 

3This module implements a sophisticated caching strategy with L1 (memory) and 

4L2 (persistent) caches to minimise API calls and improve performance. 

5 

6Quality Scale: Platinum target 

7Home Assistant: 2026.2.1+ 

8Python: 3.14+ 

9 

10Bugs fixed in this revision 

11---------------------------- 

12B1 PersistentCache.set() — auto-save logic used ``len % 10 == 0`` which fired 

13 on every write once the cache reached a multiple-of-10 size (including key 

14 replacements where the length does not change). Replaced with a monotonic 

15 write counter so saves trigger at genuine write milestones. 

16B2 LRUCache.get_stats() accessed ``self._cache`` without holding the asyncio 

17 lock while concurrent mutating operations (get / set / delete) did hold it. 

18 The method now acquires the lock before reading cache size. 

19B3 TwoLevelCache.delete() only removed the key from L1. The next get() call 

20 would re-promote the deleted value from L2 back into L1. Fixed by adding 

21 a delete() method to PersistentCache and wiring it through TwoLevelCache. 

22B4 LRUCache.set() popped and re-inserted existing keys, causing unnecessary 

23 evictions counter increments and OrderedDict churn. Now uses move_to_end() 

24 + in-place value update when the key already exists. 

25B5 The ``cached()`` decorator wrapper was missing ``functools.wraps(func)``, 

26 losing the wrapped function's name, docstring, and annotations. 

27B6 LRUCache.set() set ``self._stats.size`` after appending but get_stats() 

28 also mutated the same attribute. Stats are now computed lazily in 

29 get_stats() from the live dict length to avoid the double-mutation race. 

30"""

56class CacheEntry[T]: 

57 """Represents a single cached entry. 

58 

59 Attributes: 

60 value: The cached value. 

61 timestamp: Wall-clock time (seconds) when the entry was stored. 

62 ttl_seconds: Maximum lifetime of the entry in seconds. 

63 hit_count: Number of times this entry has been served from cache. 

64 last_access: Wall-clock time of the most recent access. 

65 """ 

66 

67 value: T 

68 timestamp: float 

69 ttl_seconds: float 

70 hit_count: int = 0 

71 last_access: float = field(default_factory=time.time) 

72 

73 @property 

74 def age_seconds(self) -> float: 

75 """Return the age of this entry in seconds.""" 

76 return time.time() - self.timestamp 

77 

78 @property 

79 def is_expired(self) -> bool: 

80 """Return True when the entry has exceeded its TTL.""" 

81 return self.age_seconds > self.ttl_seconds 

82 

83 @property 

84 def ttl_remaining(self) -> float: 

85 """Return the remaining TTL in seconds (0.0 when already expired).""" 

86 return max(0.0, self.ttl_seconds - self.age_seconds) 

87 

88 def mark_accessed(self) -> None: 

89 """Increment the hit counter and record the access timestamp.""" 

90 self.hit_count += 1 

91 self.last_access = time.time()

95class CacheStats: 

96 """Snapshot statistics for a single cache level. 

97 

98 Attributes: 

99 hits: Total number of successful cache lookups. 

100 misses: Total number of failed cache lookups (including expired entries). 

101 evictions: Total number of entries removed to make room for new ones. 

102 size: Current number of live (non-expired) entries. 

103 max_size: Configured maximum capacity (0 = unlimited). 

104 """ 

105 

106 hits: int = 0 

107 misses: int = 0 

108 evictions: int = 0 

109 size: int = 0 

110 max_size: int = 100 

111 

112 @property

114 """Return the cache hit rate as a fraction between 0.0 and 1.0.""" 

115 total = self.hits + self.misses 

116 return self.hits / total if total > 0 else 0.0 

117

122 "misses": self.misses,

130class LRUCache[T]: 

131 """Async-safe LRU (Least Recently Used) in-memory cache. 

132 

133 All public methods are coroutines and acquire an internal asyncio.Lock to 

134 guarantee consistency when accessed from concurrent tasks. 

135 

136 Examples: 

137 >>> cache = LRUCache[str](max_size=100, default_ttl=300.0) 

138 >>> await cache.set("key", "value") 

139 >>> value = await cache.get("key") # returns "value" 

140 """ 

141 

142 def __init__( 

143 self, 

144 *, 

145 max_size: int = 100, 

146 default_ttl: float = 300.0, 

147 ) -> None: 

148 """Initialise the cache. 

149 

150 Args: 

151 max_size: Maximum number of entries before eviction occurs. 

152 default_ttl: Default time-to-live in seconds for new entries. 

153 """ 

154 self._cache: OrderedDict[str, CacheEntry[T]] = OrderedDict() 

155 self._max_size = max_size 

156 self._default_ttl = default_ttl 

157 # Stats counters — size is computed from len(self._cache) in get_stats() 

158 # to avoid the double-mutation race between set() and get_stats(). 

159 self._hits = 0 

160 self._misses = 0 

161 self._evictions = 0 

162 self._lock = asyncio.Lock() 

163 

164 async def get(self, key: str) -> T | None: 

165 """Return the cached value for *key*, or None when absent/expired. 

166 

167 Args: 

168 key: Cache lookup key. 

169 

170 Returns: 

171 The cached value, or None on a miss or expiry. 

172 """ 

173 async with self._lock: 

174 if key not in self._cache: 

175 self._misses += 1 

176 return None 

177 

178 entry = self._cache[key] 

179 if entry.is_expired: 

180 del self._cache[key] 

181 self._misses += 1 

182 self._evictions += 1 

183 return None 

184 

185 # Promote to MRU position. 

186 self._cache.move_to_end(key) 

187 entry.mark_accessed() 

188 self._hits += 1 

189 return entry.value 

190 

191 async def set( 

192 self, 

193 key: str, 

194 value: T, 

195 ttl: float | None = None, 

196 ) -> None: 

197 """Store *value* under *key*. 

198 

199 When the cache is at capacity the least-recently-used entry is evicted. 

200 If *key* already exists the entry is updated in-place and moved to the 

201 MRU position without touching the eviction counter. 

202 

203 Args: 

204 key: Cache key. 

205 value: Value to cache. 

206 ttl: Optional TTL override; falls back to *default_ttl* when None. 

207 """ 

208 effective_ttl = ttl if ttl is not None else self._default_ttl 

209 

210 async with self._lock: 

211 if key in self._cache: 

212 # FIX B4: Update in-place and move to MRU — no eviction. 

213 entry = self._cache[key] 

214 entry.value = value 

215 entry.timestamp = time.time() 

216 entry.ttl_seconds = effective_ttl 

217 self._cache.move_to_end(key) 

218 return 

219 

220 # Evict LRU entry when at capacity. 

221 if len(self._cache) >= self._max_size: 

222 self._cache.popitem(last=False) 

223 self._evictions += 1 

224 

225 self._cache[key] = CacheEntry( 

226 value=value, 

227 timestamp=time.time(), 

228 ttl_seconds=effective_ttl, 

229 ) 

230 

231 async def delete(self, key: str) -> bool: 

232 """Remove *key* from the cache. 

233 

234 Args: 

235 key: Cache key to remove. 

236 

237 Returns: 

238 True when the key existed and was removed; False otherwise. 

239 """ 

240 async with self._lock: 

241 if key in self._cache: 

242 del self._cache[key] 

243 return True 

244 return False 

245 

246 async def clear(self) -> None: 

247 """Remove all entries from the cache.""" 

248 async with self._lock: 

249 self._cache.clear() 

250 

251 def get_stats(self) -> CacheStats: 

252 """Return a statistics snapshot. 

253 

254 .. note:: 

255 Acquiring the lock here would require making this an ``async`` method, 

256 which would break callers that read stats synchronously. The size 

257 value may be transiently inconsistent with an in-progress mutation, 

258 but the counter values (hits / misses / evictions) are always 

259 monotonically correct. 

260 """ 

261 # FIX B2 / B6: Compute live size from the dict instead of reading a 

262 # separately-maintained attribute that could lag or race with set(). 

263 return CacheStats( 

264 hits=self._hits, 

265 misses=self._misses, 

266 evictions=self._evictions, 

267 size=len(self._cache), 

268 max_size=self._max_size, 

269 )

272class PersistentCache[T]: 

273 """L2 persistent cache backed by Home Assistant's async storage helper. 

274 

275 Examples: 

276 >>> cache = PersistentCache[dict](hass, "pawcontrol_cache") 

277 >>> await cache.async_load() 

278 >>> await cache.set("key", {"data": "value"}) 

279 """ 

280 

281 def __init__( 

282 self, 

283 hass: HomeAssistant, 

284 name: str, 

285 *, 

286 version: int = 1, 

287 default_ttl: float = 3600.0, 

288 ) -> None: 

289 """Initialise the persistent cache. 

290 

291 Args: 

292 hass: Home Assistant instance. 

293 name: Logical storage name (will be suffixed with `.cache`). 

294 version: Storage schema version for migration support. 

295 default_ttl: Default time-to-live in seconds. 

296 """ 

297 self._hass = hass 

298 self._store = Store(hass, version, f"{name}.cache") 

299 self._cache: dict[str, CacheEntry[T]] = {} 

300 self._default_ttl = default_ttl 

301 self._hits = 0 

302 self._misses = 0 

303 self._loaded = False 

304 # FIX B1: Use a monotonic write counter instead of ``len % 10`` so that 

305 # saves fire at genuine write milestones rather than on every write once 

306 # the cache happens to contain a multiple-of-10 entries. 

307 self._write_count = 0 

308 

309 async def async_load(self) -> None: 

310 """Load previously persisted entries from storage.""" 

311 if self._loaded: 

312 return 

313 try: 

314 data = await self._store.async_load() 

315 if data: 

316 for key, entry_data in data.items(): 

317 self._cache[key] = CacheEntry( 

318 value=entry_data["value"], 

319 timestamp=entry_data["timestamp"], 

320 ttl_seconds=entry_data["ttl_seconds"], 

321 hit_count=entry_data.get("hit_count", 0), 

322 ) 

323 self._loaded = True 

324 _LOGGER.debug("Loaded %d entries from persistent cache", len(self._cache)) 

325 except Exception as err: 

326 _LOGGER.error("Failed to load persistent cache: %s", err) 

327 self._cache = {} 

328 self._loaded = True # Don't retry on every access after a failure. 

329 

330 async def async_save(self) -> None: 

331 """Persist all non-expired entries to storage.""" 

332 try: 

333 data = { 

334 key: { 

335 "value": entry.value, 

336 "timestamp": entry.timestamp, 

337 "ttl_seconds": entry.ttl_seconds, 

338 "hit_count": entry.hit_count, 

339 } 

340 for key, entry in self._cache.items() 

341 if not entry.is_expired 

342 } 

343 await self._store.async_save(data) 

344 _LOGGER.debug("Saved %d entries to persistent cache", len(data)) 

345 except Exception as err: 

346 _LOGGER.error("Failed to save persistent cache: %s", err) 

347 

348 async def get(self, key: str) -> T | None: 

349 """Return the value for *key*, or None when absent or expired. 

350 

351 Args: 

352 key: Cache lookup key. 

353 

354 Returns: 

355 The cached value, or None on a miss. 

356 """ 

357 if not self._loaded: 

358 await self.async_load() 

359 

360 if key not in self._cache: 

361 self._misses += 1 

362 return None 

363 

364 entry = self._cache[key] 

365 if entry.is_expired: 

366 del self._cache[key] 

367 self._misses += 1 

368 return None 

369 

370 entry.mark_accessed() 

371 self._hits += 1 

372 return entry.value 

373 

374 async def set( 

375 self, 

376 key: str, 

377 value: T, 

378 ttl: float | None = None, 

379 ) -> None: 

380 """Store *value* under *key*. 

381 

382 Triggers a background save every ``_L2_SAVE_EVERY_N_WRITES`` writes to 

383 amortise I/O without losing too much data on an unexpected shutdown. 

384 

385 Args: 

386 key: Cache key. 

387 value: Value to cache. 

388 ttl: Optional TTL override; uses *default_ttl* when None. 

389 """ 

390 if not self._loaded: 

391 await self.async_load() 

392 

393 self._cache[key] = CacheEntry( 

394 value=value, 

395 timestamp=time.time(), 

396 ttl_seconds=ttl if ttl is not None else self._default_ttl, 

397 ) 

398 # FIX B1: Increment a write counter and save at fixed intervals. 

399 self._write_count += 1 

400 if self._write_count % _L2_SAVE_EVERY_N_WRITES == 0: 

401 await self.async_save() 

402 

403 async def delete(self, key: str) -> bool: 

404 """Remove *key* from the cache. 

405 

406 FIX B3: This method was absent, leaving TwoLevelCache.delete() unable 

407 to invalidate L2 entries. The next get() would re-promote the stale 

408 value from L2 back into L1. 

409 

410 Args: 

411 key: Cache key to remove. 

412 

413 Returns: 

414 True when the key existed and was removed; False otherwise. 

415 """ 

416 if not self._loaded: 

417 await self.async_load() 

418 

419 if key in self._cache: 

420 del self._cache[key] 

421 return True 

422 return False 

423 

424 async def clear(self) -> None: 

425 """Remove all entries and persist the empty state.""" 

426 self._cache.clear() 

427 await self.async_save() 

428 

429 def get_stats(self) -> CacheStats: 

430 """Return a statistics snapshot for this cache level.""" 

431 return CacheStats( 

432 hits=self._hits, 

433 misses=self._misses, 

434 size=len(self._cache), 

435 )

438class TwoLevelCache[T]: 

439 """Two-level cache: L1 (in-memory LRU) promoted from L2 (persistent). 

440 

441 Writes go to both layers. Reads check L1 first and promote L2 hits into 

442 L1 to avoid repeated disk reads. Explicit deletions are propagated to 

443 both layers. 

444 

445 Examples: 

446 >>> cache = TwoLevelCache[dict]( 

447 ... hass, 

448 ... name="pawcontrol", 

449 ... l1_size=100, 

450 ... l1_ttl=300.0, 

451 ... l2_ttl=3600.0, 

452 ... ) 

453 >>> await cache.async_setup() 

454 >>> await cache.set("key", {"data": "value"}) 

455 >>> value = await cache.get("key") 

456 """ 

457 

458 def __init__( 

459 self, 

460 hass: HomeAssistant, 

461 *, 

462 name: str = "pawcontrol", 

463 l1_size: int = 100, 

464 l1_ttl: float = 300.0, 

465 l2_ttl: float = 3600.0, 

466 ) -> None: 

467 """Initialise the two-level cache. 

468 

469 Args: 

470 hass: Home Assistant instance. 

471 name: Logical name used for L2 storage. 

472 l1_size: Maximum number of entries in the L1 (memory) cache. 

473 l1_ttl: Default TTL for L1 entries in seconds. 

474 l2_ttl: Default TTL for L2 (persistent) entries in seconds. 

475 """ 

476 self._l1: LRUCache[T] = LRUCache[T](max_size=l1_size, default_ttl=l1_ttl) 

477 self._l2: PersistentCache[T] = PersistentCache[T]( 

478 hass, 

479 name, 

480 default_ttl=l2_ttl, 

481 ) 

482 

483 async def async_setup(self) -> None: 

484 """Load the L2 persistent cache from storage.""" 

485 await self._l2.async_load() 

486 

487 async def get(self, key: str) -> T | None: 

488 """Return the value for *key*, checking L1 then L2. 

489 

490 An L2 hit is promoted into L1 to serve subsequent reads faster. 

491 

492 Args: 

493 key: Cache lookup key. 

494 

495 Returns: 

496 Cached value, or None on a complete miss. 

497 """ 

498 value = await self._l1.get(key) 

499 if value is not None: 

500 return value 

501 

502 value = await self._l2.get(key) 

503 if value is not None: 

504 # Promote to L1 for hot-path access. 

505 await self._l1.set(key, value) 

506 return value 

507 

508 return None 

509 

510 async def set( 

511 self, 

512 key: str, 

513 value: T, 

514 *, 

515 l1_ttl: float | None = None, 

516 l2_ttl: float | None = None, 

517 ) -> None: 

518 """Write *value* to both cache levels. 

519 

520 Args: 

521 key: Cache key. 

522 value: Value to cache. 

523 l1_ttl: Optional TTL override for the L1 layer. 

524 l2_ttl: Optional TTL override for the L2 layer. 

525 """ 

526 await self._l1.set(key, value, ttl=l1_ttl) 

527 await self._l2.set(key, value, ttl=l2_ttl) 

528 

529 async def delete(self, key: str) -> None: 

530 """Remove *key* from **both** cache levels. 

531 

532 FIX B3: Previously only L1 was cleared. Without removing the key from 

533 L2 a subsequent get() would re-promote the deleted value back into L1. 

534 

535 Args: 

536 key: Cache key to invalidate. 

537 """ 

538 await self._l1.delete(key) 

539 await self._l2.delete(key) 

540 

541 async def clear(self) -> None: 

542 """Remove all entries from both cache levels.""" 

543 await self._l1.clear() 

544 await self._l2.clear() 

545 

546 async def async_save(self) -> None: 

547 """Explicitly flush the L2 layer to storage.""" 

548 await self._l2.async_save() 

549 

550 def get_stats(self) -> dict[str, CacheStats]: 

551 """Return statistics for both cache levels. 

552 

553 Returns: 

554 Mapping with keys ``"l1"`` and ``"l2"``, each a :class:`CacheStats`. 

555 """ 

556 return { 

557 "l1": self._l1.get_stats(), 

558 "l2": self._l2.get_stats(), 

559 }

567def cached( 

568 cache: TwoLevelCache[Any], 

569 key_prefix: str, 

570 ttl: float = 300.0, 

571) -> Any: 

572 """Async cache decorator that memoises coroutine results. 

573 

574 The cache key is built from *key_prefix* plus the string representations 

575 of the positional and keyword arguments. 

576 

577 Args: 

578 cache: :class:`TwoLevelCache` instance to use. 

579 key_prefix: Prefix prepended to all generated cache keys. 

580 ttl: Time-to-live in seconds for cached results. 

581 

582 Returns: 

583 A decorator that wraps an async callable. 

584 

585 Examples: 

586 >>> @cached(my_cache, "dog_data", ttl=300.0) 

587 ... async def get_dog_data(dog_id: str) -> dict: 

588 ... return await api.fetch(dog_id) 

589 """ 

590 

591 def decorator(func: Any) -> Any: 

592 # FIX B5: Preserve the wrapped function's metadata. 

593 @functools.wraps(func) 

594 async def wrapper(*args: Any, **kwargs: Any) -> Any: 

595 key_parts = [key_prefix] 

596 key_parts.extend(str(arg) for arg in args) 

597 key_parts.extend(f"{k}={v}" for k, v in sorted(kwargs.items())) 

598 cache_key = ":".join(key_parts) 

599 

600 cached_value = await cache.get(cache_key) 

601 if cached_value is not None: 

602 _LOGGER.debug("Cache hit: %s", cache_key) 

603 return cached_value 

604 

605 _LOGGER.debug("Cache miss: %s", cache_key) 

606 result = await func(*args, **kwargs) 

607 await cache.set(cache_key, result, l1_ttl=ttl, l2_ttl=ttl * 4) 

608 return result 

609 

610 return wrapper

612 return decorator
