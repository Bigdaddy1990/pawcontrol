1"""Shared error classification helpers for PawControl telemetry."""

3from typing import Final 

4 

5# Type alias for error classification strings 

6ErrorClassification = str 

7 

8# Mapping of explicit reason strings to classification labels 

9_REASON_CLASSIFICATIONS: Final[dict[str, ErrorClassification]] = { 

10 "missing_instance": "missing_service", 

11 "missing_services_api": "missing_service", 

12 "missing_notify_service": "missing_service", 

13 "service_not_executed": "guard_skipped", 

14 "service_unavailable": "missing_service", 

15 "auth_error": "auth_error", 

16 "authentication_error": "auth_error", 

17 "authentication_failed": "auth_error", 

18 "unauthorized": "auth_error", 

19 "forbidden": "auth_error", 

20} 

21 

22# Collections of keyword hints used to derive classifications from error messages 

23_AUTH_HINTS: Final[tuple[str, ...]] = ( 

24 "auth", 

25 "unauthorized", 

26 "forbidden", 

27 "token", 

28 "credential", 

29 "login", 

30) 

31_UNREACHABLE_HINTS: Final[tuple[str, ...]] = ( 

32 "unreachable", 

33 "not reachable", 

34 "host is down", 

35 "network is unreachable", 

36 "connection refused", 

37 "connection reset", 

38 "connection error", 

39 "device offline", 

40) 

41_TIMEOUT_HINTS: Final[tuple[str, ...]] = ( 

42 "timeout", 

43 "timed out", 

44 "deadline exceeded", 

45) 

46_RATE_LIMIT_HINTS: Final[tuple[str, ...]] = ( 

47 "rate limit", 

48 "too many requests", 

49 "429", 

50)

53def _normalise_text(value: object | None) -> str: 

54 """Return a lowercase string suitable for matching against ``value``.""" 

55 if value is None: 

56 return "" 

57 # When ``value`` is an Exception, use its string form; otherwise cast to str 

58 text = str(value) if isinstance(value, Exception) else str(value) 

59 return text.strip().lower()

62def classify_error_reason( 

63 reason: str | None, 

64 *, 

65 error: Exception | str | None = None, 

66) -> ErrorClassification: 

67 """Return a stable classification for the given error ``reason`` and ``error``. 

68 

69 This helper attempts to normalise both explicit reason strings and free-form 

70 error messages into a small set of categories, making it easier to aggregate 

71 metrics across diverse failures. It will always return one of a handful of 

72 known values such as ``auth_error``, ``device_unreachable``, ``timeout``, 

73 ``rate_limited``, ``missing_service``, ``guard_skipped``, ``exception`` or 

74 ``unknown``. 

75 """ 

76 reason_text = _normalise_text(reason) 

77 if reason_text: 

78 classified = _REASON_CLASSIFICATIONS.get(reason_text) 

79 if classified is not None: 

80 return classified 

81 error_text = _normalise_text(error) 

82 # Hints to detect authentication/authorization errors 

83 for hint in _AUTH_HINTS: 

84 if hint in error_text: 

85 return "auth_error" 

86 # Hints to detect unreachable devices or network issues 

87 for hint in _UNREACHABLE_HINTS: 

88 if hint in error_text: 

89 return "device_unreachable" 

90 # Hints to detect timeout conditions 

91 for hint in _TIMEOUT_HINTS: 

92 if hint in error_text: 

93 return "timeout" 

94 # Hints to detect rate limiting 

95 for hint in _RATE_LIMIT_HINTS: 

96 if hint in error_text: 

97 return "rate_limited" 

98 # Explicitly classify generic exception reasons 

99 if reason_text == "exception": 

100 return "exception" 

101 

102 return "unknown"
